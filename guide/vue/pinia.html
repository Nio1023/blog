                  在 actions 中的方法 可以是同步方法 也可以是异步方法

                  import { defineStore } from "pinia";

enum Names {
  TEST = "TEST",
}

type User = {
  name: string;
  age: number;
};

const result: User = {
  name: "Nio",
  age: 18,
};

const Login = (): Promise<User> => {
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve({
        name: "Nio",

  });

export const UseTestStore = defineStore(Names.TEST, {
  state: () => {
    return {
      user: <User>{},
      name: "",

  getters: {},

    // setUser() {

      const result = await Login();
      this.user = result;
      this.setName("Nio");
    },
    setName(name: string) {
      this.name = name;
    },
  },
});

                  # $subscribe

                  当 store 的数据发生改变后触发

                  // APP.vue
import { UseTestStore } from "./store";
const Test = UseTestStore();

Test.$subscribe(
  (args, state) => {
    console.log(args); // 类似于vue的watch，会有oldvalue和newvalue
    console.log(state); // 输出state的值
  },
  {
    detached: true, //可选项，当vue实例销毁时，继续监听
    deep: true, //可选项，深度监听
    flush: "post", // 执行时机 在更新DOM之后
  }
);

import { useProfileStore } from "./profileStore"; // 导入其他的store
export const usePermissionStore = defineStore("permission", {
  state() {
    return {
      list: ["add", "delete", "update", "query"], // 权限列表
    };
  },
  getters: {
    permissionList(state) {
      const profileStore = useProfileStore();
      if (profileStore.userName === "Nio") {
        return ["*"];
      } else {
        return state.list;
      }
    },
  },
});

                  # 持久化(刷新不丢失数据)

                  // main.ts
import { createApp, toRaw } from "vue";
import App from "./App.vue";
import { createPinia, PiniaPluginContext } from "pinia";

const store = createPinia();
type Options = {
  key?: string;
};
const __piniakey__: string = "Nio"; // 设置默认值
const setStorage = (key: string, value: any) => {
  localStorage.setItem(key, JSON.stringify(value));
};
const getStorage = (key: string) => {
  return localStorage.getItem(key)
    ? JSON.parse(localStorage.getItem(key) as string)
    : {};
};
const piniaPlugin = (options: Options) => {
  return (context: PiniaPluginContext) => {
    const { store } = context;
    store.$subscribe(() => {
      setStorage(
        `${options?.key ?? __piniakey__}-${store.id}`,
        toRaw(store.$state)
      );
    });
    const data = getStorage(`${options?.key ?? __piniakey__}-${store.id}`);
    return {
      ...data,
    };
  };
};
store.use(
  piniaPlugin({
    key: "pinia",
  })
);
const app = createApp(App);
app.use(store);
app.mount("#app");
